// services/bookService.js
const { Book, Author } = require("../models/init-models.js");
const { Op } = require('sequelize');

/**
 * Servicio para la gestión de libros.
 * @class BookService
 */
class BookService {
  /**
   * Recupera todos los libros con paginación y filtros opcionales.
   * @async
   * @param {number} [page=1] - Número de página actual.
   * @param {number} [limit=10] - Cantidad de libros por página.
   * @param {Object} [filters={}] - Filtros de búsqueda (search, minYear, maxYear).
   * @returns {Promise<Object>} Objeto con los datos de los libros y la información de paginación.
   */
  async getAllBooks(page = 1, limit = 10, filters = {}) {
    const offset = (page - 1) * limit;
    const where = {};

    if (filters.search) {
        where[Op.or] = [
            { titulo: { [Op.like]: `%${filters.search}%` } },
        ];
    }
    if (filters.minYear) {
        where.anyo_publicacion = { ...where.anyo_publicacion, [Op.gte]: filters.minYear };
    }
    if (filters.maxYear) {
        where.anyo_publicacion = { ...where.anyo_publicacion, [Op.lte]: filters.maxYear };
    }

    const { count, rows } = await Book.findAndCountAll({
        where,
        include: [{ model: Author, as: 'author', attributes: ['nombre'] }],
        limit,
        offset,
        order: [['titulo', 'ASC']]
    });

    return {
        data: rows.map(book => {
            const b = book.toJSON();
            b.nombre_autor = b.author ? b.author.nombre : null;
            return b;
        }),
        pagination: {
            totalItems: count,
            totalPages: Math.ceil(count / limit),
            currentPage: page,
            itemsPerPage: limit
        }
    };
  }

  /**
   * Recupera un libro por su ID.
   * @async
   * @param {number} id_libro - ID del libro a recuperar.
   * @returns {Promise<Object|null>} El libro encontrado o null si no existe.
   */
  async getBookById(id_libro) {
    const result = await Book.findByPk(id_libro, { include: [{ model: Author, as: 'author' }] });
    return result;
  }

  /**
   * Crea un nuevo libro.
   * @async
   * @param {Object} book - Objeto con los datos del libro.
   * @returns {Promise<Object>} El libro creado.
   */
  async createBook(book) {
    const result = await Book.create(book);
    return result;
  }

  /**
   * Elimina un libro por su ID.
   * @async
   * @param {number} id_libro - ID del libro a eliminar.
   * @returns {Promise<number>} Número de filas eliminadas.
   */
  async deleteBook(id_libro) {
    const numFilas = await Book.destroy({ where: { id_libro: id_libro } });
    return numFilas;
  }

  /**
   * Actualiza los datos de un libro existente.
   * @async
   * @param {Object} book - Objeto con los datos actualizados del libro (debe incluir id_libro).
   * @returns {Promise<number>} Número de filas afectadas.
   */
  async updateBook(book) {
    let numFilas = await Book.update(book, {
      where: { id_libro: book.id_libro },
    });

    if (numFilas == 0 && (await Book.findByPk(book.id_libro))) {
      numFilas = 1;
    }

    return numFilas;
  }

  /**
   * Recupera libros que tienen un número de páginas mayor o igual al especificado.
   * @async
   * @param {number} minPages - Número mínimo de páginas.
   * @returns {Promise<Array<Object>>} Lista de libros que cumplen el criterio.
   */
  async getBooksByPageCount(minPages) {
    const { count, rows } = await Book.findAndCountAll({
      where: {
        paginas: {
          [Op.gte]: minPages
        }
      },
      include: [{ model: Author, as: 'author', attributes: ['nombre'] }],
      order: [['paginas', 'DESC']]
    });

    return rows.map(book => {
      const b = book.toJSON();
      b.nombre_autor = b.author ? b.author.nombre : null;
      return b;
    });
  }
}

module.exports = new BookService();
